1. Logistic Regression

Fit a Logistic Regression using only age and chol (cholesterol) as predictors.

For a 55 year old, how high would their cholesterol need to be for the doctors to predict heart disease is present?

How high for the doctors to estimate a 90% chance that heart disease is present?

log = LogisticRegression()
log.fit(X, y)

print("Logistic Regression Model:")
print("Intercept:", log.intercept_[0])
print("Coefficients:", log.coef_[0])

log_intercept = log.intercept_[0]
coef_age1, coef_chol1 = log.coef_[0]

def find_chol(prob, age):
    log_odds = np.log(prob / (1 - prob))
    chol = (log_odds - log_intercept - coef_age1 * age) / coef_chol1
    return chol

chol_50_logreg = find_chol(0.5, 55)
chol_90_logreg = find_chol(0.9, 55)

print("Log 50% decision boundary:", chol_50_logreg)
print("Log 90% decision boundary:", chol_90_logreg)

4. Comparing Decision Boundaries

Make a scatterplot of age and chol, coloring the points by their true disease outcome. Add a line to the plot representing the linear separator (aka decision boundary) for each of the three models above.

ages = np.linspace(df['age'].min(), df['age'].max(), 100)

chol_logreg = find_chol(0.5, ages)

chol_lda = (-lda_intercept - lda_coef[0] * ages) / lda_coef[1]

chol_svc = (-svc_intercept - svc_coef[0] * ages) / svc_coef[1]

boundaries = pd.DataFrame({
    'age': np.tile(ages, 3),
    'chol': np.concatenate([chol_logreg, chol_lda, chol_svc]),
    'model': ['Logistic Regression'] * 100 + ['LDA'] * 100 + ['SVC'] * 100
})

df['diagnosis'] = df['diagnosis'].astype(str)  # Convert diagnosis to string for color coding

plot = (
    ggplot(df, aes(x='age', y='chol', color='diagnosis')) +
    geom_point(alpha=0.7) +  # Scatterplot points
    geom_line(boundaries, aes(x='age', y='chol', color='model'), size=1.2) +  # Decision boundaries
    scale_color_manual(
        values={
            "0": "blue", "1": "red",  # True outcome colors
            "Logistic Regression": "black",
            "LDA": "green",
            "SVC": "orange"
        }
    ) +
    labs(
        title="Decision Boundaries for Age and Cholesterol",
        x="Age",
        y="Cholesterol",
        color="Legend"
    ) +
    theme_minimal()
